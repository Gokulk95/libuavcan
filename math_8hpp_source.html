30  * Saturated addition where the RHS is assumed to be a positive value.
31  *
32  * This is part of the generic, promotionless, saturation math provided by this header. Ideally this would
33  * be omitted for most platforms where saturation operations were available.
34  */
35 template <
36  class SignedType,
37  typename std::enable_if<std::is_integral<SignedType>::value && std::is_signed<SignedType>::value, int>::type = 0>
38 constexpr SignedType _saturating_add_d(SignedType left, SignedType right)
39 {
40  return (left >= 0 && right > (std::numeric_limits<SignedType>::max() - left))
41  ? std::numeric_limits<SignedType>::max()
42  : static_cast<SignedType>(left + right);
43 }

45 /*
46  * Saturated subtraction where the RHS is assumed to be a positive value.
47  *
48  * This is part of the generic, promotionless, saturation math provided by this header. Ideally this would
49  * be omitted for most platforms where saturation operations were available.
50  */
51 template <
52  class SignedType,
53  typename std::enable_if<std::is_integral<SignedType>::value && std::is_signed<SignedType>::value, int>::type = 0>
54 constexpr SignedType _saturating_sub_d(SignedType left, SignedType right)
55 {
56  return (left <= 0 && right >= std::abs(std::numeric_limits<SignedType>::min() - left))
57  ? std::numeric_limits<SignedType>::min()

73 template <
74  class SignedType,
75  typename std::enable_if<std::is_integral<SignedType>::value && std::is_signed<SignedType>::value, int>::type = 0>
76 SignedType saturating_add(SignedType left, SignedType right)
77 {
78  SignedType result;
79  if (right >= 0) // make sure this is actually substraction
80  {
81  result = _saturating_add_d<SignedType>(left, right);
82  }
83  else if (right == std::numeric_limits<SignedType>::min())
84  {
85  static_assert(std::numeric_limits<SignedType>::max() + std::numeric_limits<SignedType>::min() == -1,
86  "This header assumes two's complement integer representations.");
87  // For two's compliment the abs(max) !- abs(min). This is the only
88  // place we assume two's compliment. If we need to port this to another integer representation
89  // this should be the only thing we need to fix.
90  result = _saturating_sub_d<SignedType>(left, std::numeric_limits<SignedType>::max());
91  if (result > std::numeric_limits<SignedType>::min())
92  {
93  result = static_cast<SignedType>(result - 1);
94  }
95  }
96  else
97  {
98  // Flip the sign-bit and delegate to the subtraction implementation.
99  // We assume that the integer representation can flip the bit without
100  // promotion where right > std::numeric_limits<SignedType>::min().
101  result = _saturating_sub_d<SignedType>(left, static_cast<SignedType>(-right));
102  }
103  return result;
104 }

161 template <class UnsignedType,

163  int>::type = 0>
164 constexpr UnsignedType saturating_add(UnsignedType left, UnsignedType right)
165 {
166  // Be careful here. Some ways of writing this logic will run afoul of
167  // optimizers where the compiler may assume that the result must be <
