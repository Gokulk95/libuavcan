236 TYPED_TEST_P(DurationTest, ArithmeticOperators)
237 {
238  TypeParam lhs{TypeParam::fromMicrosecond(1)};
239  TypeParam rhs{TypeParam::fromMicrosecond(1)};
240  ASSERT_EQ(TypeParam::fromMicrosecond(2), lhs += rhs);
241  lhs += TypeParam::getMaximum();
242  ASSERT_EQ(TypeParam::getMaximum(), lhs);
243  lhs -= TypeParam::fromMicrosecond(10);
244  ASSERT_EQ(TypeParam::getMaximum() - TypeParam::fromMicrosecond(10), lhs);
245  ASSERT_EQ(TypeParam::fromMicrosecond(-1), -TypeParam::fromMicrosecond(1));
246 
247  // +--[ -MAX ]------------------------------------------------------------+
248  ASSERT_EQ(TypeParam::fromMicrosecond(std::numeric_limits<typename TypeParam::MicrosecondType>::min() + 1),
249  -TypeParam::getMaximum());
250 
251  ASSERT_EQ(TypeParam::fromMicrosecond(std::numeric_limits<typename TypeParam::MicrosecondType>::min() + 2),
252  -(TypeParam::getMaximum() - TypeParam::fromMicrosecond(1)));
253 
254  // +--[ -MIN ]------------------------------------------------------------+
255  // because -MIN == MAX + 1 for twos complement integers we must assume that the minimum duration will
256  // remain saturated for -MIN
257  ASSERT_EQ(TypeParam::fromMicrosecond(std::numeric_limits<typename TypeParam::MicrosecondType>::max()),
258  -TypeParam::fromMicrosecond(std::numeric_limits<typename TypeParam::MicrosecondType>::min()));
259
